Render-smoke pattern (no HTTP — test the pure pipeline integration):
```python
from flyer_generator.postcard.schema_renderer import (
    load_template, render_postcard, PostcardContent, PostcardAddressBlock,
)
from flyer_generator.postcard.stages.pdf import assemble_postcard_pdf
from flyer_generator.stages.rasterizer import Rasterizer
import pypdf, io

@pytest.mark.parametrize("template_name", ["classic_portrait", "modern_landscape"])
@pytest.mark.parametrize("with_address", [True, False])
def test_render_smoke(template_name, with_address):
    template = load_template(template_name)
    address = (
        PostcardAddressBlock(recipient_name="Jane Doe", street="123 Main St", city_state_zip="Springfield, IL 62701")
        if with_address else None
    )
    content = PostcardContent(headline="Save the Date", body="Body copy", address_block=address)
    front_svg, back_svg = render_postcard(template, content)
    # SVG-level assertions
    assert front_svg.startswith("<svg ")
    if with_address:
        assert "Jane Doe" in back_svg
    else:
        assert "Jane Doe" not in back_svg
    # Rasterize
    rast = Rasterizer(width=template.canvas.width, height=template.canvas.height)
    front_png = rast.rasterize(front_svg)
    back_png = rast.rasterize(back_svg)
    assert front_png[:8] == b"\x89PNG\r\n\x1a\n"
    assert back_png[:8] == b"\x89PNG\r\n\x1a\n"
    # PDF round-trip
    pdf_bytes = assemble_postcard_pdf(front_png, back_png, template.canvas.width, template.canvas.height)
    reader = pypdf.PdfReader(io.BytesIO(pdf_bytes))
    assert len(reader.pages) == 2
```

```javascript
import { chromium } from "playwright";

async function fillAndSubmit(page, { template, withAddress }) {
  await page.goto("http://localhost:5173/postcards/new");
  await page.getByLabel(/headline/i).fill("Save the Date");
  await page.getByLabel(/^body$/i).fill("Body copy goes here.");
  await page.getByLabel(/template/i).fill(template);
  if (withAddress) {
    await page.getByLabel(/include recipient address block/i).check();
    await page.getByLabel(/recipient name/i).fill("Jane Doe");
    await page.getByLabel(/street/i).fill("123 Main St");
    await page.getByLabel(/city, state zip/i).fill("Springfield, IL 62701");
  }
  await page.getByRole("button", { name: /generate postcard/i }).click();
  await page.waitForURL(/\/postcards\/[A-Z0-9]{26}$/);
  const jobId = page.url().split("/").pop();
  await page.waitForFunction(
    () => /succeeded|failed/i.test(document.body.textContent ?? ""),
    { timeout: 300_000 },
  );
  // Assert 3 figure tags rendered.
  await page.waitForSelector("figure", { timeout: 10_000 });
  const figureCount = await page.locator("figure").count();
  if (figureCount < 3) {
    throw new Error(`Expected 3 figures, got ${figureCount} (jobId=${jobId})`);
  }
  return jobId;
}

const browser = await chromium.launch();
const ctx = await browser.newContext();
const page = await ctx.newPage();
const cases = [
  { template: "classic_portrait", withAddress: false },
  { template: "classic_portrait", withAddress: true },
  { template: "modern_landscape", withAddress: false },
  { template: "modern_landscape", withAddress: true },
];
for (const c of cases) {
  console.log(`Running: ${c.template} addr=${c.withAddress}`);
  const jobId = await fillAndSubmit(page, c);
  console.log(`  -> succeeded ${jobId}`);
}
await browser.close();
console.log("All 4 permutations passed");
```
