       const SLUG = /^[a-z0-9][a-z0-9-]*$/;

       type PostcardFormValues = z.infer<typeof PostcardFormSchema>;

       export function NewPostcardPage() {

         const navigate = useNavigate();

         const form = useForm<PostcardFormValues>({
           resolver: zodResolver(PostcardFormSchema),
           defaultValues: {
             headline: "",
             body: "",
             template: "classic_portrait",
             image_hint: "",
             brand_kit_slug: "",

             address_recipient_name: "",
             address_street: "",
             address_city_state_zip: "",
           },
         });

         const enqueue = useMutation({
           mutationFn: async (values: PostcardFormValues) => {
             const body: PostcardCreateRequestBody = {

               brand_kit_slug: values.brand_kit_slug || null,

             queryClient.invalidateQueries({ queryKey: queryKeys.jobs() });

             navigate(`/postcards/${job_id}`);

           onError: (err: Error) => toast.error(err.message),

                   <FormItem>
                     <FormLabel>Headline</FormLabel>
                     <FormControl><Input placeholder="Save the date" {...field} /></FormControl>
                     <FormMessage />
                   </FormItem>
                 )} />
                 <FormField control={form.control} name="body" render={({ field }) => (

                     <FormControl><Textarea rows={6} {...field} /></FormControl>
                     <FormMessage />
                   </FormItem>
                 )} />
                 <div className="grid grid-cols-2 gap-8">
                   <FormField control={form.control} name="template" render={({ field }) => (
                     <FormItem>
                       <FormLabel>Template</FormLabel>

                       <FormMessage />
                     </FormItem>
                   )} />
                   <FormField control={form.control} name="brand_kit_slug" render={({ field }) => (
                     <FormItem>
                       <FormLabel>Brand kit (optional)</FormLabel>
                       <FormControl><Input placeholder="example-co" {...field} /></FormControl>
                       <FormMessage />
                     </FormItem>
                   )} />
                 </div>
                 <FormField control={form.control} name="image_hint" render={({ field }) => (
                   <FormItem>
                     <FormLabel>Image hint (optional)</FormLabel>
                     <FormControl><Input placeholder="A field of poppies at dusk" {...field} /></FormControl>
                     <FormMessage />
                   </FormItem>
                 )} />
                 <FormField control={form.control} name="include_address" render={({ field }) => (

                     <FormControl>
                       <Switch checked={field.value} onCheckedChange={field.onChange} />
                     </FormControl>
                     <FormLabel className="!mt-0">Include recipient address block</FormLabel>
                   </FormItem>
                 )} />
                 {includeAddress && (
                   <div className="space-y-4 border-l-2 border-amber/30 pl-6">
                     <FormField control={form.control} name="address_recipient_name" render={({ field }) => (
