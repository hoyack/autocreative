Brochure renderer's element-walk pattern (we adapt to single-canvas-per-panel):

Postcard renderer is SIMPLER — no panel layout / fold lines / bleed math. Each panel == the entire canvas. The transform is `<g transform="translate(0,0)">` essentially absent. Rendering pseudocode:
```python
def render_postcard(template, content) -> tuple[str, str]:
    front_svg = _render_canvas("front", template.panels["front"], template, content)
    back_svg = _render_canvas("back", template.panels["back"], template, content)
    return front_svg, back_svg

def _render_canvas(name, panel, template, content) -> str:
    W, H = template.canvas.width, template.canvas.height
    body = ""
    if panel.background:
        defs, fill_val = render_fill(panel.background, f"bg-{name}", textures=None)
        body += defs + f'<rect x="0" y="0" width="{W}" height="{H}" fill="{fill_val}".../>'
    elements = sorted(panel.elements, key=lambda e: getattr(e, "z", 0))
    for el in elements:
        if isinstance(el, ShapeElement):
            body += render_shape(el, (0, 0, W, H), salt=...)
        elif isinstance(el, TextElement):
            body += _render_text_element(el, content, template)
        elif isinstance(el, ImagePlaceholder):
            body += _render_image_placeholder(el, content, template, images=None, salt=...)
        elif isinstance(el, LogoPlaceholder):
            body += _render_logo_placeholder(el, content, template, logo_bytes=None, salt=...)
        elif isinstance(el, BulletsElement):
            body += _render_bullets_element(el, content, template)
        elif isinstance(el, DividerElement):
            body += _render_divider(el)
    return f'<svg xmlns="http://www.w3.org/2000/svg" width="{W}" height="{H}" viewBox="0 0 {W} {H}"><rect width="{W}" height="{H}" fill="white"/>{body}</svg>'
```

Postcard PDF differs:
- Page dimensions are derived from template canvas (1200x1800 or 1800x1200), NOT a fixed bleed canvas
- No crop marks needed for v1 (postcards are typically pre-sized to USPS 4x6 / 6x4 with no bleed; postage carriers handle trim)
- Page order: front first, then back

       ```python
       """PostcardContent — runtime payload for the postcard renderer.

       Fields mirror PostcardCreateRequest (the API schema in
       flyer_generator/api/schemas/postcards.py), but are decoupled at the
       module boundary so the schema_renderer package has no dependency on the
       api/ package. Worker translates PostcardCreateRequest -> PostcardContent
       before invoking the renderer.
       """

       from __future__ import annotations

       from typing import Any

       from pydantic import BaseModel, ConfigDict, Field


       class PostcardAddressBlock(BaseModel):
           model_config = ConfigDict(extra="forbid")
           recipient_name: str = Field(min_length=1, max_length=120)
           street: str = Field(min_length=1, max_length=120)
           city_state_zip: str = Field(min_length=1, max_length=120)


       class PostcardContent(BaseModel):
           """Resolved-content payload consumed by render_postcard()."""

           model_config = ConfigDict(extra="forbid")

           headline: str = Field(min_length=1, max_length=200)
           body: str = Field(min_length=1, max_length=2000)
           image_hint: str | None = Field(default=None, max_length=500)
           address_block: PostcardAddressBlock | None = None

           def resolve_key(self, content_key: str, section_index: int | None = None) -> Any:
               """Map a template content_key to a string value (or None)."""
               if content_key == "headline":
                   return self.headline
               if content_key == "body":
                   return self.body
               if content_key == "image_hint":
                   return self.image_hint
               if content_key.startswith("address_block."):
                   suffix = content_key.removeprefix("address_block.")
                   if self.address_block is None:
                       return ""
                   return getattr(self.address_block, suffix, None)
               # Static / unknown -> None (renderer falls back to el.static_text or "").
               return None
       ```

       ```python
       def _render_canvas(
           name: str,
           panel: PanelSchema,
           template: PostcardTemplateSchema,
           content: PostcardContent,
       ) -> str:
           W = template.canvas.width
           H = template.canvas.height

           parts: list[str] = []

           if panel.background is not None:
               defs, fill_val = render_fill(panel.background, f"bg-{name}", None)
               opacity = _fill_opacity(panel.background)
               parts.append(defs)
               parts.append(
                   f'<rect x="0" y="0" width="{W}" height="{H}" '
                   f'fill="{fill_val}" fill-opacity="{opacity:.4f}"/>'
               )

           elements = sorted(panel.elements, key=lambda e: getattr(e, "z", 0))
           salt_counter = 0
           for el in elements:
               salt = f"{name}-{salt_counter}"
               salt_counter += 1

                   parts.append(_render_text_element(el, content, template))

                   parts.append(_render_logo_placeholder(el, content, template, salt))
               elif isinstance(el, DividerElement):

           page 2 is the back.

           back_panel = template.panels["back"]

           front_svg = _render_canvas("front", front_panel, template, content)
           back_svg = _render_canvas("back", back_panel, template, content)
       ```

       ```python
       Page 1 = front PNG; Page 2 = back PNG. Page dimensions are passed in by

       from reportlab.pdfgen.canvas import Canvas

       from flyer_generator.errors import RasterizationError


       class PostcardPDFError(RasterizationError):
           """Raised when postcard PDF assembly fails."""


       def assemble_postcard_pdf(
           front_png_bytes: bytes,
           back_png_bytes: bytes,
           page_width: int,
           page_height: int,
       ) -> bytes:
           """Build a 2-page PDF: page 1 = front, page 2 = back.

           Page size is supplied by the caller (template.canvas dims). Each PNG
           is drawn full-bleed to the page rect with preserveAspectRatio=False
           (caller is responsible for rasterizing at exactly page_width x
           page_height; verified by Rasterizer.rasterize's dimension check).

           Raises:
               PostcardPDFError: when either input is empty or reportlab raises.
           """
           if not front_png_bytes:
               raise PostcardPDFError("front_png_bytes must be non-empty")
           if not back_png_bytes:
               raise PostcardPDFError("back_png_bytes must be non-empty")
           if page_width <= 0 or page_height <= 0:
               raise PostcardPDFError(
                   f"page dimensions must be positive (got {page_width}x{page_height})"
               )

           buf = io.BytesIO()
           try:
               canvas = Canvas(buf, pagesize=(page_width, page_height))
               # Page 1: front
               canvas.drawImage(
                   ImageReader(io.BytesIO(front_png_bytes)),
                   0, 0,
                   width=page_width, height=page_height,
                   preserveAspectRatio=False,
               )
               canvas.showPage()
               # Page 2: back
               canvas.drawImage(
                   ImageReader(io.BytesIO(back_png_bytes)),
                   0, 0,
                   width=page_width, height=page_height,
                   preserveAspectRatio=False,
               )
               canvas.showPage()
               canvas.save()
           except Exception as exc:
               raise PostcardPDFError(f"PDF assembly failed: {exc}") from exc

           return buf.getvalue()
       ```
