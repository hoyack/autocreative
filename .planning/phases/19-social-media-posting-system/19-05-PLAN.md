    ```python
    from typing import Literal

    from pydantic import BaseModel, ConfigDict, Field

    from flyer_generator.brochure.schema_renderer.schema_model import (
        Canvas,
        LogoPlaceholder,
        Palette,
        ShapeElement,
        Typography,
    )

    _AspectString = Literal["1:1", "4:5", "1.91:1", "16:9", "9:16"]

    _TextRole = Literal[
        "title",
        "body",
        "cta",
        "caption_overlay",
        "hashtag_strip",
        "org_mark",
    ]
    _ColorRole = Literal["primary", "neutral_dark", "neutral_light", "accent"]
    _FontRole = Literal["heading", "body"]
    _FontWeight = Literal["normal", "medium", "semibold", "bold"]


    class ImageSlot(BaseModel):
        model_config = ConfigDict(extra="forbid")
        bbox: tuple[float, float, float, float]
        aspect: _AspectString
        slot_name: str = "hero"


    class TextSlot(BaseModel):
        model_config = ConfigDict(extra="forbid")
        bbox: tuple[float, float, float, float]
        role: _TextRole
        content_key: str
        max_chars: int = Field(gt=0)
        color_role: _ColorRole
        font_role: _FontRole
        font_size: int = Field(gt=0)
        font_weight: _FontWeight = "normal"
        align: Literal["left", "center", "right"] = "left"
        valign: Literal["top", "middle", "bottom"] = "top"
        uppercase: bool = False

        model_config = ConfigDict(extra="forbid")
    ```

    ```python
    """Post template JSON loader."""

    from __future__ import annotations

    import json
    from pathlib import Path

    from flyer_generator.errors import IntentUnsupportedError, PlatformUnsupportedError
    from flyer_generator.social.models import Intent, Platform
    from flyer_generator.social.schemas.schema_model import PostTemplate

    _SCHEMAS_DIR = Path(__file__).parent
    _KNOWN_PLATFORMS = {"linkedin", "twitter", "instagram", "facebook"}
    _KNOWN_INTENTS = {"announcement", "value-prop", "testimonial"}


    def load_post_template(name_or_path: str) -> PostTemplate:
        if name_or_path.endswith(".json"):
            path = Path(name_or_path)
        else:
            path = _SCHEMAS_DIR / f"{name_or_path}.json"
        if not path.exists():
            available = list_post_templates()
            raise FileNotFoundError(
                f"Post template not found: {path}. Available: {available}"
            )
        raw = json.loads(path.read_text(encoding="utf-8"))
        return PostTemplate.model_validate(raw)


    def list_post_templates() -> list[str]:
        if not _SCHEMAS_DIR.exists():
            return []
        return sorted(p.stem for p in _SCHEMAS_DIR.glob("*.json"))


    def parse_template_name(name: str) -> tuple[Platform, Intent]:
        if "__" not in name:
            raise ValueError(
                f"template name {name!r} must be '<platform>__<intent>'"
            )
        platform_part, intent_part = name.split("__", 1)
        if platform_part not in _KNOWN_PLATFORMS:
            raise PlatformUnsupportedError(
                f"unknown platform {platform_part!r} in template name {name!r}"
            )
        if intent_part not in _KNOWN_INTENTS:
            raise IntentUnsupportedError(
                f"unknown intent {intent_part!r} in template name {name!r}"
            )
        return (platform_part, intent_part)  # type: ignore[return-value]
    ```

    CRITICAL — palette and typography MUST be null in every template JSON.

    Every one of the 12 templates ships with `"palette": null` and `"typography": null` at the
    top level (or omits these keys entirely, since both fields default to None in PostTemplate).
    Brand-kit application happens at RENDER time: Plan 06 (`flyer_generator/social/renderer.py::_apply_brand_kit_to_post_template`)
    swaps the brand_kit's palette + typography into the loaded PostTemplate before the SVG
    is built. If a template JSON hard-codes colors or font families in `palette`/`typography`,
    those values override the brand kit — which defeats the entire brand-kit-per-client model
    (SOC-02, SOC-05).

    Text slots MAY reference color_role semantically ("primary", "accent", "neutral_dark",
    "neutral_light") — those strings are resolved to the brand-kit palette at render time,
    so they are safe. Likewise font_role ("heading" / "body") is resolved to the brand kit's
    heading/body family at render time. DO NOT set literal hex colors on text_slots either
    — use color_role.

    Shape fills in `shapes: [...]` are the ONE exception: shapes may carry literal hex values
    for opacity overlays (e.g. `{"type": "solid", "color": "#000000", "opacity": 0.55}` for
    a dark scrim band) because those are intentional overlays, not brand colors. The
    `SolidFill` pydantic model enforces hex validity.

    ```python
    import pytest
    from pydantic import ValidationError

    from flyer_generator.social.schemas.loader import (
        list_post_templates,
        load_post_template,
        parse_template_name,
    )

    _EXPECTED_NAMES = [

        "facebook__testimonial",
        "facebook__value-prop",
        "instagram__announcement",
        "instagram__testimonial",
        "instagram__value-prop",
        "linkedin__announcement",
        "linkedin__testimonial",
        "linkedin__value-prop",
        "twitter__announcement",
        "twitter__testimonial",
        "twitter__value-prop",
    ]

    # Platform primary canvas — (width, height). Templates may use these or a secondary aspect from their platform's rules.
    _EXPECTED_CANVASES: dict[str, set[tuple[int, int]]] = {
        "linkedin": {(1200, 627), (1200, 1200)},
        "twitter": {(1200, 675)},
        "instagram": {(1080, 1080), (1080, 1350), (1080, 1920)},
        "facebook": {(1200, 630), (1080, 1080), (1080, 1350)},
    }

        "linkedin": 3000,
        "twitter": 280,
        "instagram": 2200,
        "facebook": 63206,  # system cap; individual templates target <500 for engagement
    }


    def test_all_twelve_templates_listed() -> None:
        actual = list_post_templates()
        assert actual == _EXPECTED_NAMES, f"got {actual}"


    @pytest.mark.parametrize("name", _EXPECTED_NAMES)
    def test_template_loads_and_validates(name: str) -> None:
        t = load_post_template(name)
        assert t.schema_version == "1"
        assert t.name == name


    @pytest.mark.parametrize("name", _EXPECTED_NAMES)
    def test_template_platform_and_intent_match_filename(name: str) -> None:
        platform, intent = parse_template_name(name)
        t = load_post_template(name)
        assert t.platform == platform
        assert t.intent == intent


    @pytest.mark.parametrize("name", _EXPECTED_NAMES)
    def test_template_canvas_matches_platform_aspect(name: str) -> None:
        t = load_post_template(name)
        allowed = _EXPECTED_CANVASES[t.platform]
        actual = (t.canvas.width, t.canvas.height)
        assert actual in allowed, (
            f"{name}: canvas {actual} not in platform's allowed set {allowed}"
        )

        body_budget = t.text_budgets.get("copy.body", 0)
        cap = _MAX_BODY_CHARS[t.platform]
    ```
