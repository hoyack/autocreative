    ```python
    """Social post renderer — PostTemplate + copy + BrandKit → PNG bytes.

    Thin wrapper: reuses brochure schema_renderer SVG primitives + CairoSVG rasterization.
    Per 19-PATTERNS.md line 370: "build a one-panel PostTemplate, call the schema renderer,
    rasterize via CairoSVG, apply the Pillow crop." Here we emit the SVG inline (1 panel
    is simpler than the brochure's 6-panel loop).
    """

    from __future__ import annotations

    import base64
    import io
    from xml.sax.saxutils import escape as xml_escape

    from flyer_generator.brand_kit.models import BrandKit
    from flyer_generator.errors import SocialError
    from flyer_generator.social.models import PostCopy
    from flyer_generator.social.schemas.schema_model import PostTemplate

    _MAX_IMAGE_MP = 50_000_000


    def _apply_brand_kit_to_post_template(
        template: PostTemplate, kit: BrandKit | None
    ) -> PostTemplate:
        """Mirror Phase 18 applier.apply_brand_kit semantics for PostTemplate.

        - Palette swapped: accent from kit.palette.primary, neutral_dark/light from kit
        - Typography swapped: heading_family / body_family from kit.typography
        - Font sizes scaled by kit.size_multiplier (default 1.0)
        """
        if kit is None:
            if template.palette is None or template.typography is None:
                raise SocialError(
                    "brand_kit is required because template has null palette or typography",
                    template_name=template.name,
                )
            return template

        # Build the new Palette from kit; reuse the brochure schema_renderer Palette class
        from flyer_generator.brochure.schema_renderer.schema_model import (  # noqa: PLC0415
            Palette,
            Typography,
        )

        # Derive palette slots — fall through gracefully if kit lacks a slot
        accent = (
            kit.palette.primary.hex if kit.palette and kit.palette.primary else "#1E3A5F"
        )
        neutral_dark = (
            kit.palette.neutral_dark.hex
            if kit.palette and kit.palette.neutral_dark
            else "#1A1A1A"
        )
        neutral_light = (
            kit.palette.neutral_light.hex
            if kit.palette and kit.palette.neutral_light
            else "#FAFAF7"
        )
        muted = (
            kit.palette.accent.hex if kit.palette and kit.palette.accent else "#E8E6E1"
        )

        new_palette = Palette(
            accent_default=accent,
            neutral_dark=neutral_dark,
            neutral_light=neutral_light,
            muted=muted,
        )

        size_mult = kit.size_multiplier if kit.size_multiplier is not None else 1.0

        if kit.typography is not None:
            new_typography = Typography(
                heading_family=kit.typography.heading_family
                or "'Helvetica Neue', Helvetica, Arial, sans-serif",
                body_family=kit.typography.body_family or "Helvetica, Arial, sans-serif",
                size_scale=template.typography.size_scale
                if template.typography
                else {},
            )
        else:
            new_typography = template.typography or Typography(
                heading_family="'Helvetica Neue', Helvetica, Arial, sans-serif",
                body_family="Helvetica, Arial, sans-serif",
                size_scale={},
            )

        # Scale font_size on each text_slot by size_multiplier
        new_text_slots = [
            ts.model_copy(update={"font_size": max(1, int(round(ts.font_size * size_mult)))})
            for ts in template.text_slots
        ]

        return template.model_copy(
            update={
                "palette": new_palette,
                "typography": new_typography,
                "text_slots": new_text_slots,
            }
        )


    def _resolve_palette_color(role: str, palette) -> str:
        """Map a color_role ('primary'|'neutral_dark'|'neutral_light'|'accent') to a hex string."""
        mapping = {
            "primary": palette.accent_default,
            "accent": palette.accent_default,
            "neutral_dark": palette.neutral_dark,
            "neutral_light": palette.neutral_light,
        }
        return mapping.get(role, palette.neutral_dark)


    def _resolve_font_family(role: str, typography) -> str:
        return (
            typography.heading_family if role == "heading" else typography.body_family
        )


    def _get_content_value(copy: PostCopy, content_key: str) -> str:
        """Resolve 'copy.title' → copy.title, etc."""
        parts = content_key.split(".", 1)
        if parts[0] != "copy" or len(parts) != 2:
            return ""
        field = parts[1]
        if field == "title":
            return copy.title or ""
        if field == "body":
            return copy.body
        if field == "cta":
            return copy.cta or ""
        if field == "hashtags":
            return " ".join(copy.hashtags)
        return ""


    def _build_svg(
        template: PostTemplate,
        copy: PostCopy,
        hero_image_bytes: bytes | None,
    ) -> str:
        """Build the raw SVG string. Assumes template has non-null palette + typography
        (i.e., post-apply_brand_kit_to_post_template).
        """
        assert template.palette is not None
        assert template.typography is not None
        canvas = template.canvas
        palette = template.palette
        typography = template.typography

        # White background as default — gets covered by image_slot if present
        bg_hex = palette.neutral_light

        parts: list[str] = [
            f'<?xml version="1.0" encoding="UTF-8"?>',
            f'<svg xmlns="http://www.w3.org/2000/svg" width="{canvas.width}" height="{canvas.height}" viewBox="0 0 {canvas.width} {canvas.height}">',
            f'<rect width="{canvas.width}" height="{canvas.height}" fill="{bg_hex}"/>',
        ]

        # Image slot (z=0-ish — drawn before shapes)
        if template.image_slot is not None and hero_image_bytes is not None:
            x, y, w, h = template.image_slot.bbox
            b64 = base64.b64encode(hero_image_bytes).decode("ascii")
            parts.append(
                f'<image x="{x}" y="{y}" width="{w}" height="{h}" '
                f'preserveAspectRatio="xMidYMid slice" '
                f'xlink:href="data:image/png;base64,{b64}" '
                f'xmlns:xlink="http://www.w3.org/1999/xlink"/>'
            )

        # Shapes (sorted by z). v1: rect-only — `shape_to_svg` was never exported by
        # flyer_generator.brochure.schema_renderer.shapes (that module exposes per-kind
        # helpers like render_rect / render_rounded_rect / render_circle plus a
        # `_DISPATCH` dict + `render_shape` entry point). Phase 19 v1 only uses rect
        # shapes in its 12 templates; non-rect shapes are logged and skipped so a
        # template author accidentally using an unsupported shape does not crash
        # rendering. A future phase can widen this by importing `render_shape` and
        # passing a canvas_rect.
        import structlog  # noqa: PLC0415
        from flyer_generator.brochure.schema_renderer.shapes import (  # noqa: PLC0415
            render_rect,
        )

        _shape_log = structlog.get_logger(__name__)
        salt_counter = 0
        for shape in sorted(template.shapes, key=lambda s: getattr(s, "z", 0)):
            kind = getattr(shape, "kind", None)
            if kind != "rect":
                _shape_log.warning(
                    "social_renderer_shape_kind_unsupported_in_v1",
                    kind=kind,
                    template=template.name,
                )
                continue
            salt_counter += 1
            # render_rect(el, salt, textures=None) — textures unused in post templates v1
            parts.append(render_rect(shape, f"post-{salt_counter}"))

        # Text slots
        for ts in template.text_slots:
            text = _get_content_value(copy, ts.content_key)
            if not text:
                continue
            if ts.uppercase:
                text = text.upper()
            # XML-escape untrusted text (19-PATTERNS.md implicit from brochure precedent)
            safe = xml_escape(text)
            color = _resolve_palette_color(ts.color_role, palette)
            family = _resolve_font_family(ts.font_role, typography)
            x = ts.bbox[0]
            y = ts.bbox[1] + ts.font_size  # anchor at top-left + font_size for baseline
            anchor = {"left": "start", "center": "middle", "right": "end"}[ts.align]
            if ts.align == "center":
                x = ts.bbox[0] + ts.bbox[2] / 2
            elif ts.align == "right":
                x = ts.bbox[0] + ts.bbox[2]
            weight_map = {
                "normal": "400",
                "medium": "500",
                "semibold": "600",
                "bold": "700",
            }
            weight = weight_map.get(ts.font_weight, "400")
            parts.append(
                f'<text x="{x}" y="{y}" fill="{color}" '
                f'font-family="{family}" font-size="{ts.font_size}" '
                f'font-weight="{weight}" text-anchor="{anchor}">{safe}</text>'
            )

        parts.append("</svg>")
        return "\n".join(parts)


    def _rasterize_svg(svg: str, width: int, height: int) -> bytes:
        """Rasterize SVG string to PNG bytes via CairoSVG (primary) or resvg_py (fallback)."""
        try:
            import cairosvg  # noqa: PLC0415
            return cairosvg.svg2png(
                bytestring=svg.encode("utf-8"),
                output_width=width,
                output_height=height,
            )  # type: ignore[no-any-return]
        except ImportError:
            import resvg_py  # noqa: PLC0415
            return resvg_py.svg_to_bytes(
                svg_string=svg,
                width=width,
                height=height,
            )


    def render_post(
        template: PostTemplate,
        copy: PostCopy,
        brand_kit: BrandKit | None = None,
        *,
        hero_image_bytes: bytes | None = None,
    ) -> bytes:
        """Render a post template to PNG bytes.

        Applies brand_kit palette+typography if provided (required for templates with null palette).
        """
        w, h = template.canvas.width, template.canvas.height
        if w * h > _MAX_IMAGE_MP:
            raise SocialError(
                "canvas exceeds 50 MP cap", width=w, height=h, template=template.name
            )

        kit_applied = _apply_brand_kit_to_post_template(template, brand_kit)
        svg = _build_svg(kit_applied, copy, hero_image_bytes)
        png_bytes = _rasterize_svg(svg, w, h)
        return png_bytes
    ```
