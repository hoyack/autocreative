      const res = await fetch(`${BACKEND}/openapi.json`);
      if (!res.ok) throw new Error(`Backend not reachable: ${res.status}`);
      const j = await res.json();
      const flyerReq = j.components.schemas.FlyerCreateRequest;
      if (!flyerReq.required?.includes("template")) {
        throw new Error("Backend OpenAPI missing template field — Phase 22 backend not running");
      }
      console.log("✓ Backend ready with Phase-22 schema");
    }

    async function submitPermutation(browser, { template, subtype }) {
      const ctx = await browser.newContext({ viewport: { width: 1600, height: 1100 } });
      const page = await ctx.newPage();

      try {
        await page.goto(`${FRONTEND}/flyers/new`, { waitUntil: "networkidle" });

        // Select template via the Select component
        await page.getByTestId("template-select").click();
        await page.getByRole("option", { name: template }).click();

        // Select subtype
        await page.getByTestId("subtype-select").click();
        await page.getByRole("option", { name: subtype }).click();

        // Fill common fields
        const data = subtype === "event" ? EVENT_DATA : INFO_DATA;
        await page.getByLabel(/^title$/i).fill(data.title);
        await page.getByLabel(/^org/i).fill(data.org);
        await page.getByLabel(/style concept/i).fill(data.style_concept);

        if (subtype === "event") {
          await page.getByLabel(/^date$/i).fill(data.date);
          await page.getByLabel(/^time$/i).fill(data.time);
          await page.getByLabel(/location name/i).fill(data.location_name);
          await page.getByLabel(/location address/i).fill(data.location_address);
          await page.getByLabel(/^fees/i).fill(data.fees);
        } else {
          await page.getByLabel(/description/i).fill(data.description);
          await page.getByLabel(/call to action/i).fill(data.call_to_action);
        }

        // Submit
        await page.getByRole("button", { name: /generate|submit/i }).click();

        // Wait for status page (URL should change to /flyers/:id)
        await page.waitForURL(/\/flyers\/[a-zA-Z0-9]+/, { timeout: 10_000 });

        // Wait for rendered PNG — up to 3 minutes for a real Comfy run.
        // Look for an <img> tag within the status page that has a /renders/.../image src.
        await page.waitForSelector('img[src*="/renders/"]', { timeout: 180_000 });

        // Screenshot
        const fn = `${SHOT_DIR}/${template}-${subtype}.png`;
        await page.screenshot({ path: fn, fullPage: true });
        console.log(`✓ ${template} / ${subtype} → ${fn}`);
        return { template, subtype, status: "ok", screenshot: fn };
      } catch (err) {
        const fn = `${SHOT_DIR}/${template}-${subtype}-FAIL.png`;
        try { await page.screenshot({ path: fn, fullPage: true }); } catch {}
        console.error(`✗ ${template} / ${subtype} FAILED: ${err.message}`);
        return { template, subtype, status: "fail", error: err.message, screenshot: fn };
      } finally {
        await ctx.close();
      }
    }

    async function main() {
      await checkBackendReady();
      const browser = await chromium.launch({ headless: true });
      const results = [];
      try {
        for (const perm of PERMUTATIONS) {
          results.push(await submitPermutation(browser, perm));
        }
      } finally {
        await browser.close();
      }

      const fails = results.filter((r) => r.status === "fail");
      console.log(`\n=== SUMMARY ===`);
      console.log(`Total: ${results.length}, Passed: ${results.length - fails.length}, Failed: ${fails.length}`);
      if (fails.length) {
        for (const f of fails) console.log(`  ✗ ${f.template}/${f.subtype}: ${f.error}`);
        process.exit(1);
      }
    }

    main().catch((e) => {
      console.error(e);
      process.exit(1);
    });
