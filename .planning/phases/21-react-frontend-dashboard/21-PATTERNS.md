### `frontend/src/hooks/useJob.ts` (data-fetch hook, streaming-poll)

**Analog (cross-language):** `flyer_generator/api/routes/jobs.py::get_job` (lines 33-94).

**Why analog:** The hook's polling termination condition MUST match the server-side `JobStatus` terminal set: `{succeeded, failed, cancelled}`. That set is defined in `flyer_generator/api/models/job.py::JobStatus` (lines 22-27). Frontend hard-coding `TERMINAL` must match this enum exactly.

**Server-side enum to mirror** (`flyer_generator/api/models/job.py` lines 22-27):
```python
class JobStatus(str, enum.Enum):
    QUEUED = "queued"
    RUNNING = "running"
    SUCCEEDED = "succeeded"
    FAILED = "failed"
    CANCELLED = "cancelled"
```

**Server response shape** (`flyer_generator/api/schemas/jobs.py::JobDetail` lines 29-44 - the type the hook receives):
```python
class JobDetail(BaseModel):
    id: str
    kind: JobKind
    status: JobStatus
    started_at: datetime | None = None
    completed_at: datetime | None = None
    error_detail: dict | None = None
    result_ref: str | list[ResultLink] | None = None
    created_at: datetime
```

**Pattern to copy (verbatim from RESEARCH.md Pattern 3, lines 728-759):**
```ts
import { useQuery } from "@tanstack/react-query";
import { client, type JobDetail } from "@/api/client";

const TERMINAL = new Set(["succeeded", "failed", "cancelled"]);

export function useJob(jobId: string) {
  return useQuery<JobDetail>({
    queryKey: ["job", jobId],
    queryFn: async () => {
      const { data, error } = await client.GET("/jobs/{job_id}", {
        params: { path: { job_id: jobId } },
      });
      if (error || !data) throw new Error("failed to fetch job");
      return data;
    },
    refetchInterval: (query) => {
      const status = query.state.data?.status;
      if (status && TERMINAL.has(status)) return false;
      return 1000;
    },
    refetchIntervalInBackground: false,
    gcTime: 5 * 60 * 1000,
    staleTime: 0,
  });
}
```

**Critical correctness note:** RESEARCH.md Pitfall #1 (line 1133) - in TanStack Query v5 the `refetchInterval` callback receives `query`, not `data`. Code in this hook MUST use `query.state.data?.status` not `data?.status`.

### `frontend/src/hooks/useEnqueueFlyer.ts` (mutation hook, request-response)

**Analog (cross-language):** `flyer_generator/api/routes/flyers.py::create_flyer` (lines 17-54).

**Why analog:** This hook calls `POST /flyers`. The request body MUST satisfy `FlyerCreateRequest` (Pydantic with `ConfigDict(extra="forbid")` - any extra field 422s). The hook's success handler navigates to the polling page using the returned `JobCreated.job_id`.

**Server route signature** (`flyer_generator/api/routes/flyers.py` lines 17-54):
```python
@router.post("/flyers", status_code=202, response_model=JobCreated)
async def create_flyer(body: FlyerCreateRequest, ...) -> JobCreated:
    job_id = str(ulid.ULID())
    ...
    return JobCreated(job_id=job_id)
```

**Pattern to copy (verbatim from RESEARCH.md Pattern 2 example, lines 700-721):**
```ts
import { useMutation, useQueryClient } from "@tanstack/react-query";
import { useNavigate } from "react-router";
import { client } from "@/api/client";
import { toast } from "sonner";

export function useEnqueueFlyer() {
  const queryClient = useQueryClient();
  const navigate = useNavigate();
  return useMutation({
    mutationFn: async (body: FlyerCreateRequestBody) => {
      const { data, error, response } = await client.POST("/flyers", { body });
      if (error) {
        throw new Error(
          (error as { detail?: string })?.detail ?? `HTTP ${response.status}`,
        );
      }
      return data!;
    },
    onSuccess: ({ job_id }) => {
      queryClient.invalidateQueries({ queryKey: ["jobs"] });
      navigate(`/flyers/${job_id}`);
    },
    onError: (err) => toast.error(err.message),
  });
}
```

**Sibling hooks** (`useEnqueueBrochure`, `useEnqueuePost`, `useEnqueueCampaign`, `useEnqueueBrandKitFetch`) follow this exact shape - swap the path string, body type, and navigation target. Each maps 1:1 to `flyer_generator/api/routes/{brochures,social,brand_kits}.py`.

### `frontend/src/pages/flyers/new.tsx` (form-page, request-response)

**Analog (cross-language):** `flyer_generator/api/schemas/flyers.py::FlyerCreateRequest` (lines 14-37) + `flyer_generator/models.py::EventInput` (referenced by FlyerCreateRequest line 24).

**Why analog:** Every form field MUST mirror a Pydantic field. Constraints (`min_length`, `max_length`, `pattern`) MUST match - if zod is loose where Pydantic is strict, the user sees "valid" forms 422 from the server.

**Server-side schema (the constraint contract)** (`flyer_generator/api/schemas/flyers.py` lines 14-37):
```python
class FlyerCreateRequest(BaseModel):
    model_config = ConfigDict(extra="forbid")
    event: EventInput
    preset: str = Field(min_length=1, max_length=64)
    brand_kit_slug: str | None = Field(default=None, max_length=64)
    accent: str | None = Field(default=None, pattern=r"^#[0-9A-Fa-f]{6}$")
    max_bg_attempts: int | None = Field(default=None, ge=1, le=10)

    @field_validator("brand_kit_slug")
    @classmethod
    def _validate_slug(cls, v: str | None) -> str | None:
        if v is None: return None
        if not _SLUG_RE.match(v):  # ^[a-z0-9][a-z0-9-]*$
            raise ValueError("brand_kit_slug must match ^[a-z0-9][a-z0-9-]*$")
        return v
```

@router.get(
    "/jobs",
    response_model=PaginatedJobs,
    summary="List jobs (filter by kind / status, newest first)",
)
async def list_jobs(
    limit: int = Query(default=50, ge=1, le=200),
    offset: int = Query(default=0, ge=0),
    kind: JobKind | None = Query(default=None),
    status: JobStatus | None = Query(default=None),
    session: AsyncSession = Depends(get_session),
) -> PaginatedJobs:
    stmt = select(JobRecord).order_by(JobRecord.created_at.desc())
    count_stmt = select(func.count()).select_from(JobRecord)
    if kind is not None:
        stmt = stmt.where(JobRecord.kind == kind)
        count_stmt = count_stmt.where(JobRecord.kind == kind)
    if status is not None:
        stmt = stmt.where(JobRecord.status == status)
        count_stmt = count_stmt.where(JobRecord.status == status)

    total = (await session.execute(count_stmt)).scalar_one()
    rows = (await session.execute(stmt.limit(limit).offset(offset))).scalars().all()

    items = [
        JobDetail(  # reuse existing schema; map result_ref same way as get_job
            id=r.id, kind=r.kind, status=r.status,
            started_at=r.started_at, completed_at=r.completed_at,
            error_detail=r.error_detail,
            result_ref=(f"/api/v1/renders/{r.result_ref}/image" if r.result_ref else None),
            created_at=r.created_at,
        )
        for r in rows
    ]
    return PaginatedJobs(items=items, total=total, limit=limit, offset=offset)

**Critical security pattern to copy** (`flyer_generator/api/routes/renders.py` lines 41-59):
```python
def _is_within(candidate: Path, root: Path) -> bool:
    """True iff candidate (resolved, must exist) is inside root (resolved).
    strict=True forces the file to exist AND follows symlinks - any
    symlink-to-outside-root is rejected. is_relative_to (py >= 3.9) handles
    trailing separators correctly where str.startswith would not.
    """
    try:
        resolved = candidate.resolve(strict=True)
    except (FileNotFoundError, OSError):
        return False
    try:
        root_resolved = root.resolve(strict=True) if root.exists() else root.resolve()
    except OSError:
        return False
    return resolved.is_relative_to(root_resolved)
```

**Route shape to copy** (mirror of `renders.py` lines 62-106):
```python
@router.get(
    "/brand-kits/{slug}/logos/{filename}",
    summary="Stream a brand-kit logo file (PNG / JPG / SVG)",
)
async def get_brand_kit_logo(
    slug: str = PathParam(..., pattern=r"^[a-z0-9][a-z0-9-]*$", max_length=64),
    filename: str = PathParam(..., max_length=128),
    settings: AppSettings = Depends(get_settings),
) -> FileResponse:
    """Serve a logo from .brand-kits/{slug}/logos/{filename}.

    Returns 404 on:
    - slug or filename outside the allowed-extension whitelist
    - resolved file outside settings.brand_kits_dir / slug / logos/
    - missing file on disk
    Never distinguish which (T-1 information disclosure).
    """
    base = Path(settings.brand_kits_dir) / slug / "logos"
    candidate = base / filename
    if not _is_within(candidate, Path(settings.brand_kits_dir)):
        raise HTTPException(status_code=404, detail="logo not found")
    media_type = _ALLOWED_EXT_MIME.get(candidate.suffix.lower())
    if media_type is None:
        raise HTTPException(status_code=404, detail="logo not found")
    return FileResponse(
        path=candidate,
        media_type=media_type,
        headers={"Content-Disposition": f'inline; filename="{candidate.name}"'},
    )
```

**Reuse `_ALLOWED_EXT_MIME` from `renders.py` (lines 33-38)** OR define a logo-specific whitelist (PNG, JPG, SVG). Planner's call. Note: the renders whitelist excludes SVG; if logos can be SVG (commonly true for brand assets), the new route needs `_LOGO_EXT_MIME` with `.svg: "image/svg+xml"` added.

**Why this analog:** The threat model document attached to `renders.py` explicitly references `_validate_containment` from brand_kit/storage as the design ancestor of `_is_within`. This new route is the third generation of the same defensive pattern - the existing two are battle-tested.

## Shared Patterns

### Pattern: Pydantic `extra="forbid"` -> zod `.strict()`

**Source:** Every Phase 20 schema (`flyer_generator/api/schemas/*.py`) sets `model_config = ConfigDict(extra="forbid")`. Examples: `flyer_generator/api/schemas/brand_kits.py:18`, `schemas/flyers.py:22`, `schemas/brochures.py:17`, `schemas/social.py:21`, `schemas/jobs.py:15`.

**Apply to:** Every zod schema in `frontend/src/pages/**/*.tsx`.

**Rationale:** Pydantic 422s on unexpected fields. zod's `.strict()` mirrors this, catching frontend typos (e.g. `colour_accent` instead of `color_accent`) at form-validation time rather than as a server 422.

**Code:**
```ts
// In every form's zod schema:
const FlyerFormSchema = z.object({ /* ... */ }).strict();
```
