    // Hand-rolled ms-to-human formatter. ~5 lines, no date-fns dep.
    export function formatElapsed(ms: number): string {
      const s = Math.max(0, Math.floor(ms / 1000));
      if (s < 60) return `${s}s`;
      const m = Math.floor(s / 60);
      if (m < 60) return `${m}m ${s % 60}s`;
      const h = Math.floor(m / 60);
      return `${h}h ${m % 60}m`;
    }

The simplest approach: in tests we set `VITE_API_URL` so the client sends an absolute URL — but Vite env vars are bundle-time, not runtime. Instead, mock at the relative path:
    ```ts
    import { setupServer } from "msw/node";
    import { http, HttpResponse } from "msw";

    // openapi-fetch is configured with baseUrl "/api/v1" by default. In jsdom
    // there is no Vite proxy, so requests resolve against jsdom's default
    // location (http://localhost:3000 or similar). msw http.* handlers accept
    // a relative path and match the full URL — that's what we use.
    const BASE = "/api/v1";

    export const handlers = [
      // Default: every job is "succeeded" with a render result_ref.
      http.get(`${BASE}/jobs/:id`, ({ params }) =>
        HttpResponse.json({
          id: params.id,
          kind: "flyer",
          status: "succeeded",
          started_at: new Date().toISOString(),
          completed_at: new Date().toISOString(),
          error_detail: null,
          result_ref: "/api/v1/renders/01J9ABCDEFGHJKMNPQRSTVWXYZ/image",
          created_at: new Date().toISOString(),
        }),
      ),
      // Default: every flyer enqueue returns 202 + a deterministic job id.
      http.post(`${BASE}/flyers`, async () =>
        HttpResponse.json(
          { job_id: "01J9ABCDEFGHJKMNPQRSTVWXYZ" },
          { status: 202 },
        ),
      ),
    ];

    export const server = setupServer(...handlers);
    ```

    Per-test override: tests can call `server.use(http.get(...))` to override the default. The setup.ts `afterEach(server.resetHandlers)` restores defaults between tests.

    //                                + Pitfall #2 (gcTime, not cacheTime)
    import { useQuery } from "@tanstack/react-query";
    import { client, isTerminalStatus, type JobDetail } from "@/api/client";
    import { queryKeys } from "@/lib/queryKeys";

    export function useJob(jobId: string) {
      return useQuery<JobDetail>({
        queryKey: queryKeys.job(jobId),
        queryFn: async () => {
          const { data, error } = await client.GET("/jobs/{job_id}", {
            params: { path: { job_id: jobId } },
          });
          if (error || !data) throw new Error("failed to fetch job");
          return data;
        },
        // v5: callback receives Query object, NOT data. Read via query.state.data.
        // CRITICAL — copying a v4 example here will break polling termination.
        refetchInterval: (query) => {
          const status = query.state.data?.status;
          if (isTerminalStatus(status)) return false;
          return 1000;
        },
        refetchIntervalInBackground: false,
        gcTime: 5 * 60 * 1000,
        staleTime: 0,
      });
    }

    ```tsx
    interface RenderPreviewProps {
      url: string;
      alt?: string;
      className?: string;
    }

    /**
     * Render a single artifact URL as either an inline <img> (PNG/JPG) or
     * a download <a> (PDF). NEVER use <object> for PDFs — UX is poor across
     * browsers (per 21-RESEARCH.md "Job Polling UX" + 21-CONTEXT.md
     * <specifics> "Render preview pattern").
     */
    export function RenderPreview({ url, alt = "render", className }: RenderPreviewProps) {
      const lower = url.toLowerCase();
      if (lower.endsWith(".pdf") || lower.includes(".pdf?") || lower.includes("/pdf")) {
        const filename = url.split("/").filter(Boolean).slice(-2, -1)[0] ?? "render.pdf";
        return (
          <a
            href={url}
            download={filename}
            className={`inline-flex items-center gap-2 rounded border px-3 py-2 text-sm hover:bg-muted ${className ?? ""}`}
          >
            Download PDF
          </a>
        );
      }
      return (
        <img
          src={url}
          alt={alt}
          className={`max-w-full rounded border ${className ?? ""}`}
          loading="lazy"
        />
      );

    2. Write `frontend/src/components/JobStatusCard.tsx`. Build on RESEARCH.md Pattern 3's consumer (lines 762-797) and extend to the 5 states from CONTEXT.md "Job Polling UX":

    ```tsx
    import { useEffect, useState } from "react";
    import { useJob } from "@/hooks/useJob";
    import { Skeleton } from "@/components/ui/skeleton";
    import { Badge } from "@/components/ui/badge";
    import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
    import { RenderPreview } from "@/components/RenderPreview";
    import { formatElapsed } from "@/lib/elapsed";

    interface JobStatusCardProps {
      jobId: string;
      title?: string;
    }

    /**
     * Single source of truth for "show a job's status, then its result".
     *
     * 5 status states (per 21-CONTEXT.md "Job Polling UX"):
     *   queued    -> spinner + "Waiting in queue..."
     *   running   -> spinner + elapsed time
     *   succeeded -> RenderPreview(s) from result_ref
     *   failed    -> destructive badge + JSON-serialized error_detail
     *   cancelled -> gray "cancelled" badge
     *
     * Every status page in plans 21-06..09 renders exactly this component.
     * The brochure status page (21-07) renders a wrapper that ALSO fetches
     * the BrochureDetail (3 artifacts) — this card handles the single-render
     * case + the campaign-array case only.
     */
    export function JobStatusCard({ jobId, title }: JobStatusCardProps) {
      const { data: job, isPending, error } = useJob(jobId);

      // Live-update elapsed time while running.
      const [now, setNow] = useState<number>(() => Date.now());
      useEffect(() => {
        if (!job || job.status !== "running") return;
        const t = setInterval(() => setNow(Date.now()), 1000);
        return () => clearInterval(t);
      }, [job?.status]);

      if (isPending) return <Skeleton className="h-64 w-full" />;
      if (error) {
        return (
          <Card>
            <CardContent className="p-4 text-sm text-destructive">
              Failed to load job: {error instanceof Error ? error.message : "unknown error"}
            </CardContent>
          </Card>
        );
      }
      if (!job) return null;

      const startedMs = job.started_at ? new Date(job.started_at).getTime() : null;
      const completedMs = job.completed_at ? new Date(job.completed_at).getTime() : null;
      const elapsedMs =
        startedMs == null ? 0 : (completedMs ?? now) - startedMs;

      return (
        <Card>
          <CardHeader>
            <CardTitle className="flex items-center gap-2">
              <span>{title ?? job.kind}</span>
              <Badge
                variant={
                  job.status === "failed"
                    ? "destructive"
                    : job.status === "cancelled"
                      ? "secondary"
                      : "default"
                }
              >
                {job.status}
              </Badge>
              {job.status === "running" && (
                <span className="text-sm text-muted-foreground">
                  {formatElapsed(elapsedMs)}
                </span>
              )}
            </CardTitle>
          </CardHeader>
          <CardContent className="space-y-4">
            {job.status === "queued" && (
              <p className="text-sm text-muted-foreground">Waiting in queue...</p>
            )}
            {job.status === "running" && (
              <p className="text-sm text-muted-foreground">
                Generating... ({formatElapsed(elapsedMs)} elapsed)
              </p>
            )}
            {job.status === "succeeded" && typeof job.result_ref === "string" && (
              <RenderPreview url={job.result_ref} alt={`${job.kind} render`} />
            )}
            {job.status === "succeeded" && Array.isArray(job.result_ref) && (
              <div className="grid gap-3 sm:grid-cols-2 md:grid-cols-3">
                {job.result_ref.map((link) => (
                  <div key={link.platform} className="space-y-1">
                    <p className="text-xs font-medium text-muted-foreground">{link.platform}</p>
                    <RenderPreview url={link.url} alt={`${link.platform} render`} />
                  </div>
                ))}
              </div>
            )}
            {job.status === "failed" && job.error_detail && (
              <pre className="overflow-x-auto rounded bg-muted p-4 text-xs">
                {JSON.stringify(job.error_detail, null, 2)}
              </pre>
            )}
            {job.status === "cancelled" && (
              <p className="text-sm text-muted-foreground">This job was cancelled.</p>
            )}
          </CardContent>
        </Card>
      );
    }
    ```
