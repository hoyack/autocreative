    ```ts
    // Source: https://openapi-ts.dev/openapi-fetch/
    // Per 21-RESEARCH.md Pattern 2 + Pitfall #6: baseUrl is RELATIVE by default
    // so the Vite dev proxy in vite.config.ts forwards /api/* to localhost:8000.
    // Set VITE_API_URL only when pointing at a non-default backend.
    import createClient from "openapi-fetch";
    import type { paths } from "./schema.gen";

    export const client = createClient<paths>({
      baseUrl: import.meta.env.VITE_API_URL ?? "/api/v1",
    });

    // Re-export the root paths type so consumers can derive custom subtypes.
    export type Schemas = paths;

    // --- Request body aliases --------------------------------------------
    export type BrandKitFetchRequestBody =
      paths["/brand-kits/fetch"]["post"]["requestBody"]["content"]["application/json"];

    export type FlyerCreateRequestBody =
      paths["/flyers"]["post"]["requestBody"]["content"]["application/json"];

    export type BrochureCreateRequestBody =
      paths["/brochures"]["post"]["requestBody"]["content"]["application/json"];

    export type PostCreateRequestBody =
      paths["/social/posts"]["post"]["requestBody"]["content"]["application/json"];

    export type CampaignCreateRequestBody =
      paths["/social/campaigns"]["post"]["requestBody"]["content"]["application/json"];

    // --- Response aliases ------------------------------------------------
    export type JobCreated =
      paths["/flyers"]["post"]["responses"][202]["content"]["application/json"];

    export type JobDetail =
      paths["/jobs/{job_id}"]["get"]["responses"][200]["content"]["application/json"];

    export type PaginatedBrandKits =
      paths["/brand-kits"]["get"]["responses"][200]["content"]["application/json"];

    export type BrandKitDetail =
      paths["/brand-kits/{slug}"]["get"]["responses"][200]["content"]["application/json"];

    // --- Error body shape (stable across all 4xx/5xx, per errors.py) ----
    // Kept as a hand-written interface because openapi-typescript does not
    // currently emit a single shared type for FastAPI error responses.
    export interface ApiErrorBody {
      detail: string | unknown;
      error_type: string;
      trace_id: string;
    }

    // --- Status enum literals (mirror flyer_generator/api/models/job.py) -
    // Narrowed so `isTerminal(status)` below is typesafe.
    export const TERMINAL_STATUSES = ["succeeded", "failed", "cancelled"] as const;
    export type TerminalStatus = (typeof TERMINAL_STATUSES)[number];

    export function isTerminalStatus(s: unknown): s is TerminalStatus {
      return typeof s === "string" && (TERMINAL_STATUSES as readonly string[]).includes(s);
    }
    ```

    **Do NOT** hand-type the nested request body shapes (e.g. `FlyerCreateRequestBody`) — they come from `schema.gen.ts` via the alias. That way, any Phase 20 schema change surfaces as a TS error in the consumer instead of silent drift.

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3 (from research) | Information Disclosure | `VITE_API_URL` env var | mitigate | `baseUrl` defaults to relative `/api/v1` so no absolute backend URL is baked into the default bundle. `.env.example` (written in 21-01) warns developers NEVER to commit a production URL. |
| T-9 (new) | Tampering | Generated `schema.gen.ts` | accept | File is generated from our own backend. The snapshot is committed to git so a build-time adversary can only diff their way in by modifying committed files, which shows up in code review. |
| T-10 (new) | Tampering | `error.detail` field rendered in toasts (downstream) | mitigate | `ApiErrorBody.detail` is typed as `string | unknown`. React's default JSX escaping handles `<script>` tags in the string case. Downstream plans (21-04 onwards) MUST NOT pass `detail` through `dangerouslySetInnerHTML`. |
</threat_model>
