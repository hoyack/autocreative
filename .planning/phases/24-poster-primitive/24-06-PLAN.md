# tests/poster/test_render_smoke.py — 9 parametrized cases + dimension assertion

import io
import pytest
from PIL import Image

from flyer_generator.poster.schema_renderer.loader import load_template, list_templates
from flyer_generator.api.tasks.poster import _SIZE_TO_CANVAS
from flyer_generator.stages.composer import PosterComposer
from flyer_generator.stages.rasterizer import Rasterizer
# ... import the FlyerInput, GeneratedBackground, ResolvedLayout, VisionVerdict
# fixtures used in tests/flyer/schema_renderer/test_render_smoke.py — copy the
# minimal fixture builders (or import them).

TEMPLATES = ["editorial_grand", "bold_announcement", "cinematic_onesheet"]
SIZES = ["18x24", "24x36", "27x40"]


def _permutations():
    """Yield (template_name, size, canvas_dims) for all 9 combinations."""
    for t in TEMPLATES:
        for s in SIZES:
            yield t, s, _SIZE_TO_CANVAS[s]


@pytest.mark.parametrize(
    "template_name,size,canvas_dims",
    list(_permutations()),
    ids=lambda v: v if isinstance(v, str) else f"{v[0]}x{v[1]}",
)
def test_render_smoke_canvas_dims(template_name, size, canvas_dims):
    """For each (template × size) permutation:
       1. Load the template.
       2. Build a synthetic GeneratedBackground at canvas_dims (skip Comfy).
       3. Compose SVG via PosterComposer(canvas_width=canvas_dims[0]).
       4. Rasterize via Rasterizer(width=canvas_dims[0], height=canvas_dims[1]).
       5. Assert decoded PNG image.size == canvas_dims (Pillow round-trip).
    """
    template = load_template(template_name)
    # Build minimal FlyerInput, GeneratedBackground, ResolvedLayout, VisionVerdict
    # (see tests/flyer/schema_renderer/test_render_smoke.py for the exact shape).
    # CRITICAL: GeneratedBackground.image_bytes must be a PNG of size canvas_dims
    # (synthetic — generate a solid-color Pillow Image at canvas_dims).
    # ... fixture building omitted for brevity; mirror the flyer test pattern ...

    composer = PosterComposer(canvas_width=canvas_dims[0])
    svg = composer.compose(event, background, verdict, layout, template=template)

    rast = Rasterizer(width=canvas_dims[0], height=canvas_dims[1])
    png_bytes = rast.rasterize(svg)

    img = Image.open(io.BytesIO(png_bytes))
    assert img.size == canvas_dims, (
        f"{template_name}-{size}: expected {canvas_dims}, got {img.size}"
    )

# tests/api/test_poster_permutations.py — 9 POST happy + 3 invalid-size rejections

import pytest
from sqlalchemy import select
from flyer_generator.api.models import JobKind, JobRecord, JobStatus

TEMPLATES = ["editorial_grand", "bold_announcement", "cinematic_onesheet"]
SIZES = ["18x24", "24x36", "27x40"]
INVALID_SIZES = ["36x48", "12x18", ""]


def _permutations():
    return [(t, s) for t in TEMPLATES for s in SIZES]


@pytest.mark.parametrize("template,size", _permutations(), ids=lambda v: f"{v}")
async def test_post_poster_permutation_returns_202(client, fake_arq_pool, sessionmaker_fx, template, size):
    """All 9 (template × size) POSTs return 202 with correct enqueue."""
    payload = {
        "headline": "Phase 24 Test",
        "style_preset": "photorealistic",
        "template": template,
        "size": size,
    }
    response = await client.post("/api/v1/posters", json=payload)
    assert response.status_code == 202
    body = response.json()
    job_id = body["job_id"]
    assert len(job_id) == 26  # ULID

    async with sessionmaker_fx() as s:
        job = await s.get(JobRecord, job_id)
        assert job.kind == JobKind.POSTER
        assert job.status == JobStatus.QUEUED
        assert job.input_payload["template"] == template
        assert job.input_payload["size"] == size

    # Verify task_generate_poster was enqueued with the right payload
    assert len(fake_arq_pool.calls) == 1
    name, _args, kwargs = fake_arq_pool.calls[0]
    assert name == "task_generate_poster"
    assert kwargs["payload"]["template"] == template
    assert kwargs["payload"]["size"] == size


@pytest.mark.parametrize("invalid_size", INVALID_SIZES, ids=lambda v: f"size={v!r}")
async def test_post_poster_rejects_invalid_size(client, invalid_size):
    """3 invalid-size cases return 422 (Pydantic Literal enforcement)."""

        "style_preset": "photorealistic",
        "template": "editorial_grand",
        "size": invalid_size,
    }
    response = await client.post("/api/v1/posters", json=payload)
    assert response.status_code == 422

// /tmp/check-e2e-poster-24.mjs — Playwright harness (172+ lines, NOT committed)

import { chromium } from "playwright";

const TEMPLATES = ["editorial_grand", "bold_announcement", "cinematic_onesheet"];
const SIZES = ["18x24", "24x36", "27x40"];
const PERMUTATIONS = TEMPLATES.flatMap(t => SIZES.map(s => ({ template: t, size: s })));

const FE = process.env.FE_URL || "http://localhost:5173";
const BE = process.env.BE_URL || "http://localhost:8000";
const POLL_TIMEOUT_MS = 300_000;
const HEADLESS = process.env.HEADLESS !== "0";

// Pre-flight: assert /openapi.json has /api/v1/posters POST.
const openapi = await fetch(`${BE}/openapi.json`).then(r => r.json());
if (!openapi.paths["/api/v1/posters"]?.post) {
  console.error("BE is missing /api/v1/posters — Phase-24 backend not running?");
  process.exit(2);
}

const browser = await chromium.launch({ headless: HEADLESS });
const failures = [];

for (const { template, size } of PERMUTATIONS) {
  const context = await browser.newContext();
  const page = await context.newPage();
  try {
    await page.goto(`${FE}/posters/new`);
    await page.getByLabel(/headline/i).fill(`Phase 24 ${template} ${size}`);
    // Open size Select, click target option
    await page.getByTestId("size-select").click();
    await page.getByRole("option", { name: size }).click();
    // Open template Select, click target option
    await page.getByLabel(/template/i).click();
    await page.getByRole("option", { name: template }).click();
    await page.getByRole("button", { name: /generate poster/i }).click();
    // Wait for redirect to /posters/:id
    await page.waitForURL(/\/posters\/[A-Z0-9]{26}$/, { timeout: 30_000 });
    // Poll the page for terminal status
    const start = Date.now();
    let status = null;
    while (Date.now() - start < POLL_TIMEOUT_MS) {
      const text = await page.locator("body").innerText();
      if (/succeeded/i.test(text)) { status = "succeeded"; break; }
      if (/failed|cancelled/i.test(text)) { status = "failed"; break; }
      await page.waitForTimeout(2000);
    }
    if (status !== "succeeded") {
      failures.push({ template, size, reason: `status=${status || "timeout"}` });
      const ts = new Date().toISOString().replace(/[:.]/g, "-");
      await page.screenshot({ path: `/tmp/poster-fail-${template}-${size}-${ts}.png`, fullPage: true });
    }
  } catch (e) {
    failures.push({ template, size, reason: e.message });
    const ts = new Date().toISOString().replace(/[:.]/g, "-");
    await page.screenshot({ path: `/tmp/poster-fail-${template}-${size}-${ts}.png`, fullPage: true });
  } finally {
    await context.close();
  }
}

await browser.close();

if (failures.length === 0) {
  console.log(`All 9 permutations passed`);
  process.exit(0);
} else {
  console.error(`${failures.length} of 9 failed:`);
  for (const f of failures) console.error(`  - ${f.template}-${f.size}: ${f.reason}`);
  process.exit(1);
}
