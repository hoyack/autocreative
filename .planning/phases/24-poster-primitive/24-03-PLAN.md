# flyer_generator/api/schemas/posters.py

from __future__ import annotations
import re
from datetime import datetime
from typing import Literal
from pydantic import BaseModel, ConfigDict, Field, field_validator

_SLUG_RE = re.compile(r"^[a-z0-9][a-z0-9-]*$")

# Locked size literals per D-XX (CONTEXT decisions). Exactly 3 values.
PosterSize = Literal["18x24", "24x36", "27x40"]


class PosterCreateRequest(BaseModel):
    """Body of POST /api/v1/posters (PO-01)."""

    model_config = ConfigDict(extra="forbid")

    headline: str = Field(min_length=1, max_length=120)
    subheading: str | None = Field(default=None, max_length=200)
    cta_text: str | None = Field(default=None, max_length=120)
    image_hint: str | None = Field(default=None, max_length=500)
    brand_kit_slug: str | None = Field(default=None, max_length=64)
    style_preset: str = Field(min_length=1, max_length=64)
    template: str = Field(min_length=1, max_length=64)
    size: PosterSize

    @field_validator("brand_kit_slug")
    @classmethod
    def _validate_slug(cls, v: str | None) -> str | None:
        if v is None:
            return None
        if not _SLUG_RE.match(v):
            raise ValueError("brand_kit_slug must match ^[a-z0-9][a-z0-9-]*$")
        return v
```

# flyer_generator/api/models/poster.py

from __future__ import annotations
from datetime import datetime
from sqlalchemy import JSON, DateTime, ForeignKey, String
from sqlalchemy.orm import Mapped, mapped_column, relationship
from flyer_generator.api.models.base import Base, utcnow
from flyer_generator.api.models.render import RenderRecord


class PosterRecord(Base):
    __tablename__ = "posters"

    # Parallel-id (PO-XX): id == JobRecord.id supplied by route handler.
    # NO default=new_ulid — worker MUST set id from job_id explicitly.
    id: Mapped[str] = mapped_column(String(26), primary_key=True)
    template: Mapped[str] = mapped_column(String(64), nullable=False)
    # Stored as the size literal (e.g. "18x24") for FE display + audit.
    size: Mapped[str] = mapped_column(String(8), nullable=False)
    brand_kit_slug: Mapped[str | None] = mapped_column(
        ForeignKey("brand_kits.slug", ondelete="SET NULL"),
        nullable=True,
        index=True,
    )
    content_payload: Mapped[dict] = mapped_column(JSON, default=dict, nullable=False)
    render_id: Mapped[str | None] = mapped_column(
        ForeignKey("renders.id", ondelete="SET NULL"), nullable=True, index=True
    )
    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True), default=utcnow, nullable=False
    )

    render: Mapped[RenderRecord | None] = relationship(
        "RenderRecord", foreign_keys=[render_id], lazy="joined"
    )
```

1. **Parallel-id ORM contract verbatim**: `id: Mapped[str] = mapped_column(String(26), primary_key=True)` — NO `default=new_ulid`. Add an explicit DDL test asserting `id_col.default is None` and `id_col.server_default is None` (mirrors postcard pattern from 23-02 — that test caught a real regression risk).

2. **Postgres enum extension wrapped in autocommit_block**: `ALTER TYPE jobkind ADD VALUE IF NOT EXISTS 'POSTER'` — older Postgres rejects this inside an open txn; idempotent via `IF NOT EXISTS`.

3. **SQLite no-op for enum**: per 23-02 SUMMARY's empirical verification — the initial `0001_initial_schema` does NOT emit a CHECK constraint on `jobs.kind`. The Python-side enum extension is sufficient.

4. **Test fixture pattern for migrations**: `monkeypatch.setenv('FLYER_DATABASE_URL', async_url)` — alembic env.py:31 unconditionally overrides cfg-set sqlalchemy.url with `AppSettings.database_url`, so env-var override is the only mechanism that works. Documented in 23-02 + 22-05 SUMMARYs.

5. **PosterCreateRequest size literal MUST be exactly 3 values** — quality_gate enforces. No "ANSI/A1" deferred values are added; that's in CONTEXT.md::deferred.

6. **No `address_block`** — posters are single-canvas; no recipient block. The schema explicitly does NOT have address fields.

7. **`render.py::kind` comment update**: append `, "poster_final"` to the comment line listing valid kinds. No DDL change (kind is `String(40)` indexed, no enum).
