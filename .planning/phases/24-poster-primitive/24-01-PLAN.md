From flyer_generator/flyer/schema_renderer/loader.py (the pattern to clone):

```python
_SCHEMAS_DIR = Path(__file__).parent.parent / "schemas"

def load_template(name_or_path: str) -> FlyerTemplateSchema:
    if name_or_path.endswith(".json"):
        path = Path(name_or_path)
    else:
        path = _SCHEMAS_DIR / f"{name_or_path}.json"
    if not path.exists():
        available = list_templates()
        raise FileNotFoundError(
            f"Schema template not found: {path}. Available: {available}"
        )
    raw = json.loads(path.read_text(encoding="utf-8"))
    return FlyerTemplateSchema.model_validate(raw)

def list_templates() -> list[str]:
    if not _SCHEMAS_DIR.exists():
        return []
    return sorted(p.stem for p in _SCHEMAS_DIR.glob("*.json"))
```

The poster's loader returns `PosterTemplateSchema` instead of `FlyerTemplateSchema` and reads from `flyer_generator/poster/schemas/`. Otherwise identical.

```python
class Canvas(BaseModel):
    """Poster canvas defaults to 5400x7200 (18×24 portrait at 300 DPI)."""
    model_config = ConfigDict(extra="forbid")
    width: int = Field(default=5400, gt=0)
    height: int = Field(default=7200, gt=0)
```

Typography defaults change (print-distance reading):
```python
class Typography(BaseModel):
    model_config = ConfigDict(extra="forbid")
    heading_family: str = "'Arial Black', 'Helvetica Neue', Arial, sans-serif"
```
